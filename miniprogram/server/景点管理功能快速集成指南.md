# 景点管理功能快速集成指南

## 快速开始

### 1. 引入API
```javascript
const SpotManageApi = require('../../server/SpotManageApi')
```

### 2. 基础用法

#### 添加景点
```javascript
// 准备景点数据
const spotData = {
  name: "九寨沟",
  location: {
    address: "四川省阿坝州九寨沟县",
    geopoint: {
      latitude: 33.2640,
      longitude: 103.9237
    }
  },
  description: "九寨沟以翠海、叠瀑、彩林、雪峰、藏情、蓝冰'六绝'著称于世。",
  price: 169,
  rating: 4.7,
  opening_time: "07:00",
  closing_time: "18:00",
  tags: ["自然风光", "世界遗产", "湖泊"]
}

// 添加景点
const result = await SpotManageApi.addSpot(spotData)
```

#### 获取景点列表
```javascript
// 获取前10个景点
const result = await SpotManageApi.getSpotList({
  page: 1,
  limit: 10
})

// 搜索景点
const searchResult = await SpotManageApi.getSpotList({
  keyword: "九寨沟",
  minRating: 4.0
})
```

#### 更新景点
```javascript

  _id: "spot_123456",
```

## 完整页面示例

### 景点列表页面 (spotList.js)
```javascript
const SpotManageApi = require('../../server/SpotManageApi')

Page({
  data: {
    spotList: [],
    loading: false,
    hasMore: true,
    currentPage: 0
  },

  onLoad() {
    this.loadSpotList()
  },

  // 加载景点列表
  async loadSpotList() {
    if (this.data.loading) return
    
    this.setData({ loading: true })

    try {
      const result = await SpotManageApi.getSpotList({
        page: this.data.currentPage + 1,
        limit: 10
      })

      if (result.success) {
        this.setData({
          spotList: this.data.currentPage === 0 ? result.data : [...this.data.spotList, ...result.data],
          currentPage: this.data.currentPage + 1,
          hasMore: result.data.length === 10
        })
      } else {
        wx.showToast({
          title: result.message,
          icon: 'error'
        })
      }
    } catch (error) {
      wx.showToast({
        title: '加载失败',
        icon: 'error'
      })
    } finally {
      this.setData({ loading: false })
    }
  },

  // 下拉刷新
  onPullDownRefresh() {
    this.setData({
      spotList: [],
      currentPage: 0,
      hasMore: true
    })
    this.loadSpotList().then(() => {
      wx.stopPullDownRefresh()
    })
  },

  // 上拉加载更多
  onReachBottom() {
    if (this.data.hasMore) {
      this.loadSpotList()
    }
  },

  // 查看景点详情
  viewSpotDetail(e) {
    const spotId = e.currentTarget.dataset.id
    wx.navigateTo({
      url: `/pages/spotDetail/spotDetail?id=${spotId}`
    })
  }
})
```

### 景点详情页面 (spotDetail.js)
```javascript
const SpotManageApi = require('../../server/SpotManageApi')

Page({
  data: {
    spotDetail: null,
    loading: true
  },

  onLoad(options) {
    if (options.id) {
      this.loadSpotDetail(options.id)
    }
  },

  // 加载景点详情
  async loadSpotDetail(spotId) {
    try {
      const result = await SpotManageApi.getSpot(spotId)
      
      if (result.success) {
        this.setData({
          spotDetail: result.data,
          loading: false
        })
      } else {
        wx.showToast({
          title: result.message,
          icon: 'error'
        })
      }
    } catch (error) {
      wx.showToast({
        title: '加载失败',
        icon: 'error'
      })
    }
  },

  // 打开地图
  openMap() {
    const { location } = this.data.spotDetail
    wx.openLocation({
      latitude: location.geopoint.latitude,
      longitude: location.geopoint.longitude,
      name: this.data.spotDetail.name,
      address: location.address
    })
  },

  // 拨打电话
  makeCall() {
    wx.makePhoneCall({
      phoneNumber: this.data.spotDetail.contact_phone
    })
  }
})
```

### 添加景点页面 (addSpot.js)
```javascript
const SpotManageApi = require('../../server/SpotManageApi')

Page({
  data: {
    formData: {
      name: '',
      location: {
        address: '',
        geopoint: null
      },
      description: '',
      price: 0,
      opening_time: '08:00',
      closing_time: '18:00',
      contact_phone: '',
      website: '',
      tags: [],
      facilities: []
    },
    submitting: false
  },

  // 输入处理
  onInputChange(e) {
    const { field } = e.currentTarget.dataset
    const { value } = e.detail
    
    this.setData({
      [`formData.${field}`]: value
    })
  },

  // 选择位置
  chooseLocation() {
    wx.chooseLocation({
      success: (res) => {
        this.setData({
          'formData.location.address': res.address,
          'formData.location.geopoint': {
            latitude: res.latitude,
            longitude: res.longitude
          }
        })
      }
    })
  },

  // 上传图片
  uploadImages() {
    wx.chooseImage({
      count: 5,
      success: async (res) => {
        wx.showLoading({ title: '上传中...' })
        
        try {
          const uploadPromises = res.tempFilePaths.map(filePath => {
            return wx.cloud.uploadFile({
              cloudPath: `spots/${Date.now()}_${Math.random()}.jpg`,
              filePath
            })
          })
          
          const uploadResults = await Promise.all(uploadPromises)
          const imageUrls = uploadResults.map(result => result.fileID)
          
          this.setData({
            'formData.images': [...(this.data.formData.images || []), ...imageUrls]
          })
          
          wx.hideLoading()
          wx.showToast({
            title: '上传成功',
            icon: 'success'
          })
        } catch (error) {
          wx.hideLoading()
          wx.showToast({
            title: '上传失败',
            icon: 'error'
          })
        }
      }
    })
  },

  // 提交表单
  async submitForm() {
    if (this.data.submitting) return

    // 数据验证
    const validation = SpotManageApi.validateSpotData(this.data.formData)
    if (!validation.isValid) {
      wx.showModal({
        title: '请检查输入信息',
        content: validation.errors.join('\n'),
        showCancel: false
      })
      return
    }

    this.setData({ submitting: true })

    try {
      const result = await SpotManageApi.addSpot(this.data.formData)
      
      if (result.success) {
        wx.showToast({
          title: '添加成功',
          icon: 'success'
        })
        
        setTimeout(() => {
          wx.navigateBack()
        }, 1500)
      } else {
        wx.showToast({
          title: result.message,
          icon: 'error'
        })
      }
    } catch (error) {
      wx.showToast({
        title: '添加失败',
        icon: 'error'
      })
    } finally {
      this.setData({ submitting: false })
    }
  }
})
```

## 搜索功能增强

```javascript
// 在景点列表页面添加搜索功能
Page({
  data: {
    searchKeyword: '',
    searchTimer: null
  },

  // 搜索输入
  onSearchInput(e) {
    const keyword = e.detail.value
    this.setData({ searchKeyword: keyword })

    // 防抖搜索
    if (this.data.searchTimer) {
      clearTimeout(this.data.searchTimer)
    }

    this.data.searchTimer = setTimeout(() => {
      this.searchSpots(keyword)
    }, 500)
  },

  // 搜索景点
  async searchSpots(keyword = this.data.searchKeyword) {
    this.setData({
      spotList: [],
      currentPage: 0,
      loading: true
    })

    try {
      const result = await SpotManageApi.getSpotList({
        page: 1,
        limit: 10,
        keyword: keyword
      })

      if (result.success) {
        this.setData({
          spotList: result.data,
          currentPage: 1,
          hasMore: result.data.length === 10
        })
      }
    } catch (error) {
      wx.showToast({
        title: '搜索失败',
        icon: 'error'
      })
    } finally {
      this.setData({ loading: false })
    }
  }
})
```

## 常用工具函数

```javascript
// 格式化价格显示
function formatPrice(price) {
  if (price === 0) return '免费'
  return `¥${price}`
}

// 格式化评分显示
function formatRating(rating) {
  return `${rating.toFixed(1)}分`
}

// 计算距离（需要用户位置）
function calculateDistance(lat1, lon1, lat2, lon2) {
  const R = 6371 // 地球半径（公里）
  const dLat = (lat2 - lat1) * Math.PI / 180
  const dLon = (lon2 - lon1) * Math.PI / 180
  const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
    Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
    Math.sin(dLon/2) * Math.sin(dLon/2)
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a))
  return R * c
}

module.exports = {
  formatPrice,
  formatRating,
  calculateDistance
}
```
